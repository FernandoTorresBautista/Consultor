router.get('/ui/:nombre?/:sessionID?', function(req, res) {
	UserModel.getUi(req.params.nombre, req.sessionID, function(error, data){
		if( !(data && data.msg) ){

				if( !(data2 && data2.msg) ){

					res.send(data2[0].content);
				}
				else{
					res.json(403,{"msg":""});
				}
			});

		}
		else{
			res.json(403,{"msg":""});
		}
	})
});
